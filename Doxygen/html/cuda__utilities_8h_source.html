58  inline uint trailing_right ( uint n ) {
59  uint c = 32; // c will be the number of zero bits on the right
60  n &= -signed(n);

64  if (n & 0x0F0F0F0F) c -= 4;
65  if (n & 0x33333333) c -= 2;
